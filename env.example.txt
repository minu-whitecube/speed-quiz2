# Supabase 설정
# 이 파일을 .env.local로 복사하고 실제 값으로 변경하세요
# Supabase 프로젝트의 URL과 Anon Key를 입력하세요
# https://supabase.com/dashboard/project/YOUR_PROJECT/settings/api 에서 확인할 수 있습니다

NEXT_PUBLIC_SUPABASE_URL=your_supabase_project_url
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_supabase_anon_key

# Google Analytics 4 설정

NEXT_PUBLIC_GA_MEASUREMENT_ID=G-C47B9E0XMP
